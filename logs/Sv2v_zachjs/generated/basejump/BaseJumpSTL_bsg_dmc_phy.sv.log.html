module tx_bit_slice (
	clk_2x_n,
	wrdata_en_90,
	wrdata_90,
	dq_o,
	dq_oe_n
);
	input clk_2x_n;
	input wrdata_en_90;
	input [1:0] wrdata_90;
	output reg dq_o;
	output reg dq_oe_n;
	reg odd;
	always @(posedge clk_2x_n)
		if (wrdata_en_90) begin
			odd <= ~odd;
			if (odd)
				dq_o <= wrdata_90[1];
			else
				dq_o <= wrdata_90[0];
			dq_oe_n <= 1'b0;
		end
		else begin
			odd <= 1'b0;
			dq_oe_n <= 1'b1;
		end
endmodule
